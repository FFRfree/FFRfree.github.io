      import numpy as np
  a,b,m_expression = input('输入下限，上限，函数    ').       split        ()
  a = eval(a)
  b = eval(b)
  segs = eval(input('输入精度'))
  count = 0
  area = 0
  for x in np.linspace(a,b,segs):
      aprx_area = exec(m_expression) *(b-a)
      area += aprx_area
      count += 1
  output = area/count
  print(output)
